--SCRIPT PESQUISAR PROFISSIONAIS ATIVOS CADASTRADOS NO CNES
select 	prof.nome_prof "NOME", prof.data_nasc "DATA NASCIMENTO", 
	CASE WHEN PROF.CD_PAIS = '' THEN 'BRASIL' ELSE PAIS.DS_PAIS END "NACIONALIDADE", 
	PROF.NUM_IDENT "RG", prof.cpf_prof "CPF", dt_entrada "ENTRADA CNES", equipe.cod_area || ' - ' || EQUIPE2.NM_REFERENCIA "EQUIPE", 
	equipe.microarea "MICRO AREA", CBO.COD_CBO ||' - ' || cbo.descricao "CBO", CONSELHO.DESCRICAO "CONSELHO", 
	REGISTRO.N_REGISTRO "NUM CONSELHO", REGISTRO.SG_UF_CRM "UF", prof.cod_cns "CNS", PROF.NO_EMAIL "EMAIL"
	from cnes.lfces018 prof
inner join cnes.lfces038 equipe on prof.prof_id = equipe.prof_id
left join cnes.nfces026 CBO on cbo.cod_cbo = equipe.cod_cbo
LEFT JOIN CNES.NFCES036 PAIS ON PAIS.CD_PAIS = PROF.CD_PAIS
LEFT JOIN CNES.LFCES021 REGISTRO ON REGISTRO.PROF_ID = PROF.PROF_ID
LEFT JOIN CNES.NFCES033 CONSELHO ON CONSELHO.CODORGEMIS = REGISTRO.CONSELHOID
LEFT JOIN CNES.LFCES037 EQUIPE2 ON EQUIPE2.COD_AREA = EQUIPE.COD_AREA
WHERE EQUIPE.DT_DESLIGAMENTO IS NULL
order by EQUIPE.COD_AREA
