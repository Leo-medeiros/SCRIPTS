SELECT 

	VP.PESSOA_ID,V.ID,VP.DATA,V.NOME,V.faixa_de_idade,V.DOENCAS_PREVENIDAS, 
	V.DOSE,P.NOME, CASE P.SEXO
	WHEN 'F' THEN 'FEMINO'
	WHEN 'T' THEN 'MASCULINO'
	ELSE '' END AS SEXO, TO_CHAR(P.DATA_NASCIMENTO, 'TMday, DD "de" TMMonth YYYY') AS DT_NASC
FROM siabfacil.vacina_pessoa VP
LEFT JOIN SIABFACIL.VACINA V ON (VP.VACINA_ID = V.ID)
LEFT JOIN SIABFACIL.PESSOA P ON (VP.PESSOA_ID = P.ID)
